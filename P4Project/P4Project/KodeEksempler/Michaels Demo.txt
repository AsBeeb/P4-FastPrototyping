struct vaaben {
	local string navn;
	local int skade;
	local int skadeType;
	vaaben (string Navn, int Skade, int SkadeType){
		navn = Navn;
		skade = Skade;
		skadeType = SkadeType;
	}
}

struct karakter {
	local string navn;
	local int liv;
	local vaaben udstyr;
	karakter (string Navn, int Liv, Vaaben Udstyr){
		navn = Navn;
		liv = Liv;
		udstyr = Udstyr;
	}
}

func void Main ()
{
	local vaaben stridsOekse = vaaben ("Tordenskrald", 15, 1);
	local vaaben trylleStav  = vaaben ("Nattens Vogtoer", 13, 3);
	local karakter michael = karakter("Michael", 50, stridsOekse);
	local karakter moerketsFyrste = karakter ("Moerkets Fyrste", 70, tryllestav);

	local karakter[] karakterListe = michael + moerketsFyrste;
	play (spiller vs modstander in karakterListe){
	Angrib (spiller, modstander);

	} until (moerketsFyrste.liv < 0 || michael.liv < 0);
}

func void Angrib(karakter angriber, karakter forsvarrer){
	forsvarrer.liv = forsvarrer.liv - angriber.udstyr.skade;
}
